353 static paxg::Vec2f calculateCatmullRom(const paxg::Vec2f& p0, const paxg::Vec2f& p1,
354 const paxg::Vec2f& p2, const paxg::Vec2f& p3, float t) {
355 float t2 = t * t;
356 float t3 = t2 * t;
357
358 float v0 = (p2.x() - p0.x()) * 0.5f;
359 float v1 = (p3.x() - p1.x()) * 0.5f;
360 float x = (2 * p1.x() - 2 * p2.x() + v0 + v1) * t3 + (-3 * p1.x() + 3 * p2.x() - 2 * v0 - v1) * t2 + v0 * t + p1.x();
361
362 v0 = (p2.y() - p0.y()) * 0.5f;
363 v1 = (p3.y() - p1.y()) * 0.5f;
364 float y = (2 * p1.y() - 2 * p2.y() + v0 + v1) * t3 + (-3 * p1.y() + 3 * p2.y() - 2 * v0 - v1) * t2 + v0 * t + p1.y();

503 static paxg::Vec2f calculateTangent(const paxg::Vec2f& p0, const paxg::Vec2f& p1,
504 const paxg::Vec2f& p2, const paxg::Vec2f& p3, float t) {
505 const float t2 = t * t;
506
507 // Catmull-Romã‚¹ãƒ—ãƒ©ã‚¤ãƒ³ã®1æ¬¡å¾®åˆ†
508 // First derivative of Catmull-Rom spline
509 const float v0_x = (p2.x() - p0.x()) * 0.5f;
510 const float v1_x = (p3.x() - p1.x()) * 0.5f;
511 const float dx = (2 * p1.x() - 2 * p2.x() + v0_x + v1_x) * 3.0f * t2 +
512 (-3 * p1.x() + 3 * p2.x() - 2 * v0_x - v1_x) * 2.0f * t + v0_x;
513
514 const float v0_y = (p2.y() - p0.y()) * 0.5f;
515 const float v1_y = (p3.y() - p1.y()) * 0.5f;
516 const float dy = (2 * p1.y() - 2 * p2.y() + v0_y + v1_y) * 3.0f * t2 +
517 (-3 * p1.y() + 3 * p2.y() - 2 * v0_y - v1_y) * 2.0f * t + v0_y;
518
519 // æ­£è¦åŒ–ï¼ˆå˜ä½ãƒ™ã‚¯ãƒˆãƒ«åŒ–ï¼‰
520 // Normalize to unit vector
521 const float length = std::sqrt(dx * dx + dy * dy);
522 if (length > 0.0001f) {
523 return paxg::Vec2f(dx / length, dy / length);
524 }
525 return paxg::Vec2f(1.0f, 0.0f); // ãƒ‡ãƒ•ã‚©ãƒ«ãƒˆæ–¹å‘ / Default direction
526 }
