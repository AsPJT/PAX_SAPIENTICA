107 std::cout << "Loading " << name << " data..." << std::endl;

113 PAXS_ERROR("No files are found: " + file_path);
114 return;
115 }
116
117 if(file_names[0].find(".tsv") != std::string::npos) {
118 loadNumericTSV(file_path);
119 } else if(file_names[0].find(".txt") != std::string::npos) {
120 if (z_mag > 1) loadNumericTextAndCompress(file_names);
121 else loadNumericText(file_names);
122 } else if(file_names[0].find(".bin") != std::string::npos) {
123 if constexpr (std::is_same<DataType, std::uint_least8_t>::value) {
124 loadBinary<paxs::Input8BitBinary>(file_path);
125 } else if constexpr (std::is_same<DataType, std::int_least16_t>::value) {
126 loadBinary<paxs::Input16BitBinary>(file_path);

129 PAXS_WARNING("File type is invalid: " + file_names[0]);

137 const Vector2 start_position = SimulationConstants::getInstance()->getStartArea() * z_mag;
138 const Vector2 end_position = SimulationConstants::getInstance()->getEndArea() * z_mag;
139
140 const Vector2 start_xyz_position = SimulationConstants::getInstance()->getStartArea() * pixel_size * z_mag;
141
142 std::uint_least32_t file_count = (end_position.x - start_position.x + 1) * (end_position.y - start_position.y + 1);
143 std::uint_least32_t load_count = 0;
144
145 for (GridType y = start_position.y; y <= end_position.y; ++y) {
146 for (GridType x = start_position.x; x <= end_position.x; ++x) {
147 const std::string file_name = "zxy_" + std::to_string(default_z) + "_" + std::to_string(x) + "_" + std::to_string(y) + ".bin";
148
149 if (!std::filesystem::exists(AppConfig::getInstance()->getRootPath() + file_path + file_name)) {
150 --file_count;
151 StatusDisplayer::displayProgressBar(load_count, file_count);
152 continue;
153 }
154
155 const Vector2 default_position = Vector2(x, y) * pixel_size - start_xyz_position;
156 BinaryDataType bi(file_path + file_name, AppConfig::getInstance()->getRootPath());
157 bi.calc([&](const DataType value_, const std::uint_least8_t px, const std::uint_least8_t py) {
158 const Vector2 position = default_position + Vector2((GridType)px, (GridType)py);
159 data[position.to(DataGridsType{})] = value_;
160 }
161 );
162
163 ++load_count;
164 StatusDisplayer::displayProgressBar(load_count, file_count);
165 }
166 }
167
168 std::cout << std::endl << "Loading " << name << " is completed." << std::endl;
169 std::cout << load_count << " files are loaded.\n" << std::endl;
