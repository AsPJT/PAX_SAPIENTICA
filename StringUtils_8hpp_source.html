143 static void replace(std::string& str, const std::string& from, const std::string& to) noexcept {
144 if (from.empty()) return;
145 const std::size_t from_len = from.length();
146 const std::size_t to_len = to.length();
147 std::size_t pos = 0;
148 while ((pos = str.find(from, pos)) != std::string::npos) {
149 str.replace(pos, from_len, to);
150 pos += to_len;
151 }
152 }
