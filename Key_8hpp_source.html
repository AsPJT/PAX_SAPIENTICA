283 Coordinate(double movement_size, const paxs::MercatorDeg& coordinate_) : movement_size(movement_size), coordinate(coordinate_) {
284 increase_x_keys.resize(2);
285 increase_x_keys[0] = (Key(SIV3D_KEY_A));
286 increase_x_keys[1] = (Key(SIV3D_KEY_LEFT));

288 decrease_x_keys.resize(2);

292 increase_y_keys.resize(2);

294 increase_y_keys[1] = (Key(SIV3D_KEY_DOWN));
295
296 decrease_y_keys.resize(2);
297 decrease_y_keys[0] = (Key(SIV3D_KEY_W));
298 decrease_y_keys[1] = (Key(SIV3D_KEY_UP));
299 }
300 void update(const double width) {
301 if (pressed(increase_x_keys)) increase_coordinate(coordinate.x, width);
302 if (pressed(decrease_x_keys)) decrease_coordinate(coordinate.x, width);
303 if (pressed(increase_y_keys)) increase_coordinate(coordinate.y, width);
304 if (pressed(decrease_y_keys)) decrease_coordinate(coordinate.y, width);
305 }
306 double getX() const {
307 return coordinate.x;
308 }
309 double getY() const {
310 return coordinate.y;
311 }
312 void setX(const double x_) {
313 coordinate.x = x_;
314 }
315 void setY(const double y_) {
316 coordinate.y = y_;
317 }
318
319 double toEquirectangularRadY() const {
320 return coordinate.toEquirectangularRadY();
321 }
322 double toEquirectangularDegY() const {
323 return coordinate.toEquirectangularDegY();
324 }
325
326 private:
327 paxs::MercatorDeg coordinate; // „Éû„ÉÉ„ÉóÂ∫ßÊ®ô„ÅÆ‰∏≠Â§Æ
328 std::vector<Key> increase_x_keys;
329 std::vector<Key> decrease_x_keys;
330 std::vector<Key> increase_y_keys;
331 std::vector<Key> decrease_y_keys;
332
333 void increase_coordinate(double& value, const double width) {
334 value -= (width / movement_size);
335 if (value < -180.0) {
336 value += 360.0;
337 }
338 }
339 void decrease_coordinate(double& value, const double width) {
340 value += (width / movement_size);
341 if (value >= 180.0) {
342 value -= 360.0;
343 }
344 }
