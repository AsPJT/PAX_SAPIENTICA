184 class Key {
185 public:
186 Key() = default;
187 explicit Key(InputKey key) : key(key) {}
188 bool isPressed() const {
189 return key.pressed();
190 }
191
194 static bool isShiftPressed() {
195#if defined(PAXS_USING_SIV3D)
196 return s3d::KeyShift.pressed();
197#elif defined(PAXS_USING_DXLIB)
198 return (DxLib::CheckHitKey(KEY_INPUT_LSHIFT) == 1 || DxLib::CheckHitKey(KEY_INPUT_RSHIFT) == 1);
199#elif defined(PAXS_USING_SFML)
200 return (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::LShift) || sf::Keyboard::isKeyPressed(sf::Keyboard::Key::RShift));
201#else
202 return false;
203#endif
204 }
205
208 static bool isCtrlPressed() {
209#if defined(PAXS_USING_SIV3D)
210 return s3d::KeyControl.pressed();
211#elif defined(PAXS_USING_DXLIB)
212 return (DxLib::CheckHitKey(KEY_INPUT_LCONTROL) == 1 || DxLib::CheckHitKey(KEY_INPUT_RCONTROL) == 1);
213#elif defined(PAXS_USING_SFML)
214 return (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::LControl) || sf::Keyboard::isKeyPressed(sf::Keyboard::Key::RControl));
215#else
216 return false;
217#endif
218 }
219
222 static bool isAltPressed() {
223#if defined(PAXS_USING_SIV3D)
224 return s3d::KeyAlt.pressed();
225#elif defined(PAXS_USING_DXLIB)
226 return (DxLib::CheckHitKey(KEY_INPUT_LALT) == 1 || DxLib::CheckHitKey(KEY_INPUT_RALT) == 1);
227#elif defined(PAXS_USING_SFML)
228 return (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::LAlt) || sf::Keyboard::isKeyPressed(sf::Keyboard::Key::RAlt));
229#else
230 return false;
231#endif
232 }
233
236 static bool isCommandPressed() {
237#if defined(PAXS_USING_SIV3D)
238 return s3d::KeyCommand.pressed();
239#elif defined(PAXS_USING_DXLIB)
240 // DxLib does not support Command key
241 return false;
242#elif defined(PAXS_USING_SFML)
243 return (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::LSystem) || sf::Keyboard::isKeyPressed(sf::Keyboard::Key::RSystem));
244#else
245 return false;
246#endif
247 }
248
249 private:
250 InputKey key{};
251 };

254 bool pressed(const std::array<Key, N>& keys) {
255 for (const auto& key : keys) {
