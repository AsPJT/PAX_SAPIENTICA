145 static constexpr std::uint_least32_t calcHash(const char* const str, const std::uint_least32_t seed = 0) {
146 std::uint_least32_t hash = seed;
147 std::size_t length = 0;
148 for (;; length += MM3::chunk_size) {
149 if (str[length] == '\0') {
150 break;
151 }
152 if (str[length + 1] == '\0') {
153 hash = murmur3b(str + length, 1, hash);
154 length += 1;
155 break;
156 }
157 if (str[length + 2] == '\0') {
158 hash = murmur3b(str + length, 2, hash);
159 length += 2;
160 break;
161 }
162 if (str[length + 3] == '\0') {
163 hash = murmur3b(str + length, 3, hash);
164 length += 3;
165 break;
166 }
167 hash = murmur3_Chunk(str + length, hash);
168 }
169 return murmur3c(hash, length);
