96 bool existPoint(const AStarVec2& vec2_, const double cost_) noexcept {
97 auto i = std::find(closed.begin(), closed.end(), vec2_);
98 if (i != closed.end()) {
99 if ((*i).cost + (*i).distance < cost_) {
100 return true;
101 }
102 closed.erase(i);
103 return false;
104 }
105 i = std::find(open.begin(), open.end(), vec2_);

107 return false;

109 if ((*i).cost + (*i).distance < cost_) {
110 return true;
111 }
112 open.erase(i);

116 double calcCost(std::shared_ptr<Environment>& environment, const AStarVec2& neighbour_z) const noexcept {
117 // Êµ∑„ÅÆÈÄöË°å„Ç≥„Çπ„Éà
118 if (!environment->isLand(neighbour_z)) {
119 return SimulationConstants::getInstance().ocean_cost;
120 }
121 // Êµ∑Â≤∏„ÅÆÈÄöË°å„Ç≥„Çπ„Éà
122 if (environment->isCoast(neighbour_z)) {
123 return SimulationConstants::getInstance().coast_cost;
124 }
125 const double land_cost = SimulationConstants::getInstance().land_cost;
126 if (environment->getSlope(neighbour_z) < 6 /*0.1*/) return land_cost;
127 if (environment->getSlope(neighbour_z) < 100/*5*/) return 1.8 / 1.5 * land_cost;
128 if (environment->getSlope(neighbour_z) < 133/*10*/) return 2.1 / 1.5 * land_cost;
129 if (environment->getSlope(neighbour_z) < 154/*15*/) return 2.5 / 1.5 * land_cost;
130 if (environment->getSlope(neighbour_z) < 169/*20*/) return 3.0 / 1.5 * land_cost;
131 if (environment->getSlope(neighbour_z) < 181/*25*/) return 3.6 / 1.5 * land_cost;
132 if (environment->getSlope(neighbour_z) < 191/*30*/) return 4.3 / 1.5 * land_cost;
133 if (environment->getSlope(neighbour_z) < 199/*35*/) return 5.1 / 1.5 * land_cost;
134 if (environment->getSlope(neighbour_z) < 206/*40*/) return 6.0 / 1.5 * land_cost;
135 if (environment->getSlope(neighbour_z) <= 250/*90*/) return 100 * 34.7 / 1.5 * land_cost;
136
137 return SimulationConstants::getInstance().land_cost;
138 }

140 bool cellOpen(const AStarNode& node_, std::shared_ptr<Environment>& environment) noexcept {
141 for (std::int_fast32_t x{}; x < 8; ++x) {
142 const AStarVec2 neighbour = AStarVec2{ node_.position.x + astar_adjacent_cell[x].x,
143 node_.position.y + astar_adjacent_cell[x].y };
144 if (neighbour == end_vec2) return true;
145
146 const AStarVec2 neighbour_z = neighbour * z;
147 if (!search_range.contains(neighbour)) continue;
148 // „Ç≥„Çπ„ÉàË®àÁÆó
149 const double node_cost = calcCost(environment, neighbour_z) * ((x <= 3 /* Êñú„ÇÅÁßªÂãï */) ? MathConstants<double>::sqrt2() : 1.0) + node_.cost;
150 const GridType distance = calculateDistance(neighbour);
151 if (existPoint(neighbour, node_cost + distance)) continue;
152 open.emplace_back(AStarNode(neighbour, node_.position, distance, node_cost));
153 }
154 return false;
155 }

157 bool search(std::shared_ptr<Environment>& environment) noexcept {
158 open.emplace_back(AStarNode(start_vec2, AStarVec2{}, calculateDistance(start_vec2), 0));
159 while (!open.empty()) {
160 const AStarNode nn{ open.front() };
161 open.pop_front();
162 closed.emplace_back(nn);
163 if (cellOpen(nn, environment)) return true;
164 }
165 return false;
166 }

179 AStarVec2 parent_node{ closed.back().parent_node };
180 for (auto&& i{ closed.rbegin() }; i != closed.rend(); ++i) {
181 if (!((*i).position == parent_node) || ((*i).position == start_vec2)) continue;
182 path_.emplace_back((*i).position * z + sub);
183 parent_node = (*i).parent_node;
184 }

186 }
