104  void moveSettlement(const std::uint_least32_t id, const Vector2 current_key, const Vector2 target_key) noexcept {
105  auto it = settlement_grids.find(target_key.toU64());
106  if (it != settlement_grids.end()) {

111  settlement_grid.moveSettlementToThis(settlement_grids[current_key.toU64()].getSettlement(id));

119  void step() noexcept {
120  std::vector<std::tuple<std::uint_least32_t, Vector2, Vector2>> move_list;
121  for (auto& settlement_grid : settlement_grids) {
122  std::vector<Settlement>& settlements = settlement_grid.second.getSettlements();
123  for (std::size_t i = 0; i < settlements.size(); ++i) {
124  if (settlements[i].isMoved()) {
125  continue;
126  }
127 
128  auto [target_id, current_key, target_key] = settlements[i].move(gen, move_probability);
129 
130  if (target_id != 0) {
131  move_list.emplace_back(target_id, current_key, target_key);
132  }
133  }
134  }
135 
136  for (auto& move : move_list) {
137  moveSettlement(std::get<0>(move), std::get<1>(move), std::get<2>(move));
138  }
