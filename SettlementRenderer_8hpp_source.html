127 static void drawSettlements(
128 const paxs::UnorderedMap<SettlementGridsType, paxs::SettlementGrid>* agents,
129 const Vector2<double>& map_view_size,
130 const Vector2<double>& map_view_center,
131 const std::size_t select_draw
132 ) {
133 for (const auto& agent : *agents) {
134 for (const auto& settlement : agent.second.cgetSettlements()) {
135 const auto coordinate = positionToWebMercator(settlement.getPosition());
136
137 // çµŒç·¯åº¦ã®ç¯„å›²å¤–ã‚’é™¤å»
138 if (!isInViewport(coordinate, map_view_size, map_view_center)) continue;
139
140 // æç”»ä½ç½®
141 const paxg::Vec2<double> draw_pos = MapCoordinateConverter::toScreenPos(
142 coordinate,
143 map_view_size,
144 map_view_center
145 );
146
147 // å††ã®ã‚µã‚¤ã‚º
148 const float circle_size = CIRCLE_BASE_SIZE +
149 (settlement.getPopulation() / CIRCLE_POPULATION_SCALE);
150
151 // è¡¨ç¤ºãƒ¢ãƒ¼ãƒ‰åˆ¥ã«æç”»
152 if (select_draw != 5) {
153 // äººå£ã€è¾²è€•ã€mtDNAã€SNPã€é’éŠ…å™¨
154 double pop_original = 0.0;
155 switch (select_draw) {
156 case 1:
157 pop_original = static_cast<double>(settlement.getPopulation());
158 break;
159 case 2:
160 pop_original = settlement.getFarmingPopulation() /
161 static_cast<double>(settlement.getPopulation()) * MAX_POPULATION_VISUALIZATION;
162 break;
163 case 3:
164 pop_original = settlement.getMostMtDNA() /
165 MTDNA_SCALE_FACTOR * MAX_POPULATION_VISUALIZATION;
166 break;
167 case 4:
168 pop_original = settlement.getSNP() * MAX_POPULATION_VISUALIZATION;
169 break;
170 case 6:
171 pop_original = static_cast<double>(settlement.getBronze());
172 break;
173 }
174
175 const std::uint_least8_t pop = (pop_original >= MAX_POPULATION_VISUALIZATION)
176 ? static_cast<std::uint_least8_t>(MAX_POPULATION_VISUALIZATION)
177 : static_cast<std::uint_least8_t>(pop_original);
178 paxg::Circle(draw_pos, circle_size).draw(
179 SimulationColor::getSettlementColor(pop));
180 }
181 else {
182 // è¨€èª
183 paxg::Circle(draw_pos, circle_size).draw(
184 getLanguageColor(settlement.getLanguage()));
185 }
186 }
187 }

323 static void drawGridLines(
324 const Vector2<double>& map_view_size,
325 const Vector2<double>& map_view_center
326 ) {
327 const auto area_width = SimulationConstants::getInstance().getEndArea().x -
328 SimulationConstants::getInstance().getStartArea().x;
329 const auto area_height = SimulationConstants::getInstance().getEndArea().y -
330 SimulationConstants::getInstance().getStartArea().y;
331
332 const paxs::WebMercatorDeg start_coordinate = positionToWebMercator(paxs::Vector2<int>(0, 0));
333 const paxg::Vec2f draw_start_pos = paxg::Vec2f{
334 static_cast<float>((start_coordinate.x - (map_view_center.x - map_view_size.x / 2)) /
335 map_view_size.x * double(paxg::Window::width())),
336 static_cast<float>(double(paxg::Window::height()) -
337 ((start_coordinate.y - (map_view_center.y - map_view_size.y / 2)) /
338 map_view_size.y * double(paxg::Window::height())))
339 };
340
341 const paxs::WebMercatorDeg end_coordinate = positionToWebMercator(
342 paxs::Vector2<int>(area_width * 256, area_height * 256));
343 const paxg::Vec2f draw_end_pos = paxg::Vec2f{
344 static_cast<float>((end_coordinate.x - (map_view_center.x - map_view_size.x / 2)) /
345 map_view_size.x * double(paxg::Window::width())),
346 static_cast<float>(double(paxg::Window::height()) -
347 ((end_coordinate.y - (map_view_center.y - map_view_size.y / 2)) /
348 map_view_size.y * double(paxg::Window::height())))
349 };
350
351 const paxs::WebMercatorDeg tile_coordinate = positionToWebMercator(
352 paxs::Vector2<int>(SimulationConstants::getInstance().cell_group_length,
353 SimulationConstants::getInstance().cell_group_length));
354 const paxg::Vec2f tile_pos = paxg::Vec2f{
355 static_cast<float>((tile_coordinate.x - (map_view_center.x - map_view_size.x / 2)) /
356 map_view_size.x * double(paxg::Window::width())) - draw_start_pos.x(),
357 static_cast<float>(double(paxg::Window::height()) -
358 ((tile_coordinate.y - (map_view_center.y - map_view_size.y / 2)) /
359 map_view_size.y * double(paxg::Window::height()))) - draw_start_pos.y()
360 };
361
362 // å¤–æ ç·šã‚’æç”»
363 paxg::Line(draw_start_pos.x(), draw_start_pos.y(),
364 draw_start_pos.x(), draw_end_pos.y()).draw(GRID_OUTER_LINE_WIDTH, paxg::Color(0, 0, 0));
365 paxg::Line(draw_start_pos.x(), draw_start_pos.y(),
366 draw_end_pos.x(), draw_start_pos.y()).draw(GRID_OUTER_LINE_WIDTH, paxg::Color(0, 0, 0));
367 paxg::Line(draw_end_pos.x(), draw_start_pos.y(),
368 draw_end_pos.x(), draw_end_pos.y()).draw(GRID_OUTER_LINE_WIDTH, paxg::Color(0, 0, 0));
369 paxg::Line(draw_start_pos.x(), draw_end_pos.y(),
370 draw_end_pos.x(), draw_end_pos.y()).draw(GRID_OUTER_LINE_WIDTH, paxg::Color(0, 0, 0));
371
372 // å‚ç›´ã‚°ãƒªãƒƒãƒ‰ç·š
373 for (float i = draw_start_pos.x(); i < draw_end_pos.x(); i += tile_pos.x()) {
374 paxg::Line(i, draw_start_pos.y(), i, draw_end_pos.y()).draw(
375 GRID_INNER_LINE_WIDTH, paxg::Color(0, 0, 0));
376 }
377 // æ°´å¹³ã‚°ãƒªãƒƒãƒ‰ç·š
378 for (float i = draw_start_pos.y(); i < draw_end_pos.y(); i += tile_pos.y()) {
379 paxg::Line(draw_start_pos.x(), i, draw_end_pos.x(), i).draw(
380 GRID_INNER_LINE_WIDTH, paxg::Color(0, 0, 0));
381 }
382 }
