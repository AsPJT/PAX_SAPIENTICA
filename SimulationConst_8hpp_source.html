257 std::string path = "";
258 AppConfig::getInstance()->calcDataSettings(MurMur3::calcHash("SimulationProvincesPath"),
259 [&](const std::string& path_) {path = path_; });
260 // Sample „ÇíÈÅ∏Êäû„É¢„Éá„É´Âêç„Å´ÁΩÆÊèõ
261 paxs::StringExtensions::replace(path, "Sample", model_name_);
262 path += std::string("/" + life_span_file);
263
264 paxs::InputFile life_span_tsv(path);
265 if (life_span_tsv.fail()) {
266 PAXS_WARNING("Failed to read LifeSpan TSV file: " + path);
267 return;
268 }
269 // 1 Ë°åÁõÆ„ÇíË™≠„ÅøËæº„ÇÄ
270 if (!(life_span_tsv.getLine())) {
271 return; // ‰Ωï„ÇÇ„Å™„ÅÑÂ†¥Âêà

273 // BOM „ÇíÂâäÈô§
274 life_span_tsv.deleteBOM();
275 // 1 Ë°åÁõÆ„ÇíÂàÜÂâ≤„Åô„Çã
276 std::unordered_map<std::uint_least32_t, std::size_t> menu = life_span_tsv.splitHashMapMurMur3('\t');
277#ifdef PAXS_DEVELOPMENT
278 std::size_t i = 1;
279#endif
280 // ÁîüÂëΩË°®„ÇíÂàùÊúüÂåñ
281 life_span = lifeSpan{};
282 // 1 Ë°å„Åö„Å§Ë™≠„ÅøËæº„ÅøÔºàÂå∫Âàá„Çä„ÅØ„Çø„ÉñÔºâ
283 while (life_span_tsv.getLine()) {
284 std::vector<std::string> sub_menu_v = life_span_tsv.split('\t');
285 if (
286 sub_menu_v.size() <= getMenuIndex(menu, MurMur3::calcHash("hunter_gatherer_male_ndx")) ||
287 sub_menu_v.size() <= getMenuIndex(menu, MurMur3::calcHash("hunter_gatherer_female_ndx")) ||
288 sub_menu_v.size() <= getMenuIndex(menu, MurMur3::calcHash("agricultural_male_ndx")) ||
289 sub_menu_v.size() <= getMenuIndex(menu, MurMur3::calcHash("agricultural_female_ndx"))
290 ) {
291#ifdef PAXS_DEVELOPMENT
292 PAXS_WARNING("Failed to read Japan LifeSpan TSV file: " + path + " at line " + std::to_string(i));
293#endif
294
295 life_span.weight_farming_female.emplace_back(0.0);
296 life_span.weight_farming_male.emplace_back(0.0);

303 }
