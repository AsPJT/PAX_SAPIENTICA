191 void drawInternal(float x, float y, float w, float h, float r, const sf::Color& c) const
192 {
193 r = std::min(r, w * 0.5f);
194 r = std::min(r, h * 0.5f);
195
196 if (r <= 0.f) {
197 sf::RectangleShape rect({w, h});
198 rect.setPosition({x, y});
199 rect.setFillColor(c);
200 Window::window().draw(rect);
201 return;
202 }
203
204 // ÂçäÂæÑ„Å´Âøú„Åò„Å¶ÂàÜÂâ≤Êï∞„ÇíÊ±∫„ÇÅ„Çã
205 const unsigned cornerPointCount = calcCornerPointCount(r);
206 const unsigned totalPoints = 4 * (cornerPointCount + 1);
207
208 sf::ConvexShape shape;
209 shape.setPointCount(totalPoints);
210 shape.setFillColor(c);
211
212 const float left = x;
213 const float top = y;
214 const float right = x + w;
215 const float bottom = y + h;
216
217 const sf::Vector2f c_tl(left + r, top + r);
218 const sf::Vector2f c_tr(right - r, top + r);
219 const sf::Vector2f c_br(right - r, bottom - r);
220 const sf::Vector2f c_bl(left + r, bottom - r);
221
222 auto putCorner = [&](unsigned startIndex,
223 const sf::Vector2f& center,
224 float startDeg)
225 {
226 for (unsigned i = 0; i <= cornerPointCount; ++i) {
227 float ang = startDeg + 90.f * (static_cast<float>(i) / cornerPointCount);
228 float rad = paxs::Math<float>::degToRad(ang);
229 float px = center.x + std::cos(rad) * r;
230 float py = center.y + std::sin(rad) * r;
231 shape.setPoint(startIndex + i, {px, py});
232 }
233 };
234
235 unsigned idx = 0;
236 putCorner(idx, c_tl, paxs::Math<float>::deg180());
237 idx += cornerPointCount + 1;
238 putCorner(idx, c_tr, paxs::Math<float>::deg270());
239 idx += cornerPointCount + 1;
240 putCorner(idx, c_br, paxs::Math<float>::deg0());
241 idx += cornerPointCount + 1;
242 putCorner(idx, c_bl, paxs::Math<float>::deg90());
243
244 Window::window().draw(shape);
245 }
