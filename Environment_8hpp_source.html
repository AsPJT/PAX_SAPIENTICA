70 void loadFromFile(const std::string& setting_file_path) noexcept {
71 std::vector<std::vector<std::string>> settings;
72 settings = FileSystem::readTSV(setting_file_path);
73
74 if (settings.empty()) {

80 int key_column = -1;
81 int data_type_column = -1;
82 int file_path_column = -1;
83 int z_column = -1;
84 for (std::size_t i = 0; i < settings[0].size(); ++i) {
85 if (settings[0][i] == "key") {
86 key_column = int(i);
87 }
88 if (settings[0][i] == "data_type") {
89 data_type_column = int(i);
90 }
91 if (settings[0][i] == "file_path") {

93 }
94 if (settings[0][i] == "z") {
95 z_column = int(i);

97 }
98 if (key_column == -1 || data_type_column == -1 || file_path_column == -1 || z_column == -1) {
99 PAXS_WARNING("key or data_type or file_path or z is not found in " + setting_file_path);
100 return;
101 }
102
103 for (std::size_t i = 1; i < settings.size(); ++i) {
104 // ÂûãÂêç„Çístring„Åã„ÇâT„Å´Â§âÊèõ
105 const std::uint_least32_t data_type = MurMur3::calcHash(settings[i][data_type_column].size(), settings[i][data_type_column].c_str());
106 const std::uint_least32_t key = MurMur3::calcHash(settings[i][key_column].size(), settings[i][key_column].c_str());
107 const std::string& key_str = settings[i][key_column];
108 const int z_value = StringUtils::safeStoi(settings[i][z_column], 0, true);
109
110 if (data_type == MurMur3::calcHash("u8")) {
111 data_map.emplace(key, std::make_unique<DataVariant>(Data<std::uint_least8_t>(settings[i][file_path_column], key_str, z_value, progress_reporter_)));
112 }
113 else if (data_type == MurMur3::calcHash("u32")) {
114 data_map.emplace(key, std::make_unique<DataVariant>(Data<std::uint_least32_t>(settings[i][file_path_column], key_str, z_value, progress_reporter_)));
115 }
116 else if (data_type == MurMur3::calcHash("f32")) {
117 data_map.emplace(key, std::make_unique<DataVariant>(Data<float>(settings[i][file_path_column], key_str, z_value, progress_reporter_)));
118 }
119 else if (data_type == MurMur3::calcHash("s16")) {
120 data_map.emplace(key, std::make_unique<DataVariant>(Data<std::int_least16_t>(settings[i][file_path_column], key_str, z_value, progress_reporter_)));
121 }
122 else {
123 PAXS_WARNING("data_type is not found in " + setting_file_path);
124 }
125 }
