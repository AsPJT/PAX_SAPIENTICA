68 bool load(const std::string& relative_path) {
69 if (is_loaded_) {
70 PAXS_WARNING("TsvTable already loaded. Ignoring duplicate load.");
71 return is_successfully_loaded_;
72 }
73 is_loaded_ = true;
74
75 paxs::InputFile input_file(relative_path);
76 if (input_file.fail()) {
77 PAXS_ERROR(relative_path + " could not be opened.");
78 return false;
79 }
80
81 // Read header row
82 if (!input_file.getLine()) {
83 PAXS_ERROR("The first line (header) of " + relative_path + " could not be read.");
84 return false;
85 }
86
87 // Delete BOM if present
88 input_file.deleteBOM();
89
90 // Parse header
91 header_map_ = input_file.splitHashMapMurMur3('\t');
92 header_keys_ = input_file.splitHashMapMurMur3Vector('\t');
93
94 if (header_keys_.empty()) {
95 PAXS_ERROR(relative_path + " has an empty header row.");
96 return false;
97 }
98
99 // Read data rows
100 while (input_file.getLine()) {
101 std::vector<std::string> row = input_file.split('\t');
102 if (!row.empty()) {
103 rows_.emplace_back(std::move(row));
104 }
105 }
106
107 is_successfully_loaded_ = true;
108 return true;
109 }
