120 void inputLanguage_Region(const std::string& japan_provinces_path) noexcept {
121 const std::string path = japan_provinces_path + "/Language.tsv";
122 paxs::TsvTable table(path);
123
124 if (!table.isSuccessfullyLoaded()) {
125 PAXS_WARNING("Failed to read Language TSV file: " + path);
126 return;
127 }
128
129 static const std::uint_least32_t language_region_hash = MurMur3::calcHash("language_region");
130 static const std::uint_least32_t language_dist_hash = MurMur3::calcHash("language_dist");
131
132 if (!table.hasColumn(language_region_hash) || !table.hasColumn(language_dist_hash)) {
133 PAXS_ERROR("Language.tsv: missing required columns");
134 return;
135 }
136
137 for (std::size_t row = 0; row < table.rowCount(); ++row) {
138 mtDNA_Region language_region;
139 std::vector<std::string> dist = paxs::StringUtils::split(table.get(row, language_dist_hash), '/');
140
141 if (dist.size() % 2 == 1 || dist.size() <= 1) {
142 continue;
143 }
144
145 for (std::size_t j = 0; j < dist.size(); j += 2) {
146 for (std::size_t k = 0; k < language_list.size(); ++k) {
147 if (language_list[k] == dist[j]) {
148 language_region.id.emplace_back(static_cast<std::uint_least8_t>(k));
149 language_region.weight.emplace_back(StringUtils::safeStod(dist[j + 1], 0.0, true));
150 break;
151 }
152 }
153 }
154
155 if (language_region.weight.empty()) {
156 continue;
157 }
158
159 language_region.dist = std::discrete_distribution<>(language_region.weight.begin(), language_region.weight.end());
160
161 const std::string& region_str = table.get(row, language_region_hash);
162 language_region_list.emplace(MurMur3::calcHash(region_str.size(), region_str.c_str()), language_region);
163 }
