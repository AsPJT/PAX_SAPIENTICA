75 void update(const double map_view_center_x, const double map_view_center_y, const double map_view_width, const double map_view_height) {
76 for (std::size_t i = 0; i < location_range_list.size(); ++i) {
77 auto& jj = location_range_list[i];
78 if (jj.mex > map_view_center_x - map_view_width / 2 &&
79 jj.msx < map_view_center_x + map_view_width / 2 &&
80 jj.mey > map_view_center_y - map_view_height / 2 &&
81 jj.msy < map_view_center_y + map_view_height / 2) {
82 // テクスチャを描画する
83 jj.texture.resizedDrawAt( // 描画時に指定した画像サイズへ変換
84 paxg::Vec2i(
85 static_cast<int>(jj.mx / map_view_width * double(paxg::Window::width()))
86 , static_cast<int>(jj.my / map_view_height * double(paxg::Window::height()))// * 1.3
87 )
88 , // 指定位置を画像の中心として描画
89 paxg::Vec2i(
90 static_cast<int>((jj.mcx - (map_view_center_x - map_view_width / 2)) / map_view_width * double(paxg::Window::width()) - 0),
91 static_cast<int>(double(paxg::Window::height()) - ((jj.mcy - (map_view_center_y - map_view_height / 2)) / map_view_height * double(paxg::Window::height())))// + 270
92 ));
93 }
94 }
