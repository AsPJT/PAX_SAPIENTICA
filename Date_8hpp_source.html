87 // 年月日
88 class IslamicDate {
89 private:
90 DateYear year{};
91 DateMonth month{};
92 DateDay day{};
93 public:
94 // ヒジュラ暦の日付から絶対年代（日付）を計算
95 constexpr operator int() const {
96 return (day + 29 * (month - 1) + month / 2
97 + 354 * (year - 1) // 前年の閏日以外の日
98 + (3 + (11 * year)) / 30 // 前年の閏日
99 + 227014); // カレンダーの開始日の前日
100 }

102 IslamicDate() = default;
103 IslamicDate(const DateYear year_, const DateMonth month_, const DateDay day_)
104 :year(year_), month(month_), day(day_) {}
