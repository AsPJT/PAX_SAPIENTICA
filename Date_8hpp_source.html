67 template<typename T>
68 concept DateType = requires(const T date) {
69 { date.getDay() } -> std::convertible_to<std::int_least64_t>;
70 { date.isLeapMonth() } -> std::convertible_to<bool>;
71 { date.getDateOutputType() } -> std::same_as<DateOutputType>;
72 };

75 template<typename T>
76 concept YearMonthDayDateType = DateType<T> && requires(const T date) {
77 { date.getYear() } -> std::convertible_to<DateYear>;
78 { date.getMonth() } -> std::convertible_to<DateMonth>;
