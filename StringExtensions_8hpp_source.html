196 static void replace(std::string& str, const std::string& from, const std::string& to) noexcept {
197 if (from.empty()) return;
198 const std::size_t from_len = from.length();
199 const std::size_t to_len = to.length();
200 std::size_t pos = 0;
201 while ((pos = str.find(from, pos)) != std::string::npos) {
202 str.replace(pos, from_len, to);
203 pos += to_len;
204 }
205 }
