191 ZoomState calculateZoom(double map_view_height) const {

256 const std::string z_value = std::to_string(z); // Z ã®å€¤

263 // ãƒ•ã‚¡ã‚¤ãƒ«ã‚’ä¿å­˜
264 for (int i = start_cell.y, k = 0; i <= end_cell.y; ++i) {

266 const std::string y_value = std::to_string((i + z_num) & (z_num - 1)); // Y ã®å€¤
267 std::string local_file_path_zny = local_file_path_zn;
268 paxs::StringUtils::replace(local_file_path_zny, "{y}", y_value);
269
270 const std::uint_least64_t index_zy = tile_cache_.encodeKeyY(static_cast<std::uint_least64_t>((i + z_num) & (z_num - 1))) + index_z;
271
272 // URL ã¨ Binary ã®ãƒ‘ã‚¹ã‚’ Y ç½®æ›ã¾ã§äº‹å‰è¨ˆç®—ï¼ˆX ãƒ«ãƒ¼ãƒ—å¤–ï¼‰

274 std::string binary_path_zny;

276 if (!metadata_.texture_url.empty()) {
277 url_path_zny = metadata_.texture_url;
278 paxs::StringUtils::replace(url_path_zny, "{z}", z_value);
279 paxs::StringUtils::replace(url_path_zny, "{y}", y_value);
280 }
281
282 if (!binary_file_name_format.empty()) {
283 binary_path_zny = binary_file_name_format;
284 paxs::StringUtils::replace(binary_path_zny, "{z}", z_value);
285 paxs::StringUtils::replace(binary_path_zny, "{y}", y_value);
286 if (map_name.size() != 0) paxs::StringUtils::replace(binary_path_zny, "{n}", map_name);

289 for (int j = start_cell.x; j <= end_cell.x; ++j, ++k) {

291 // ç”»åƒã‚’æ ¼ç´ã™ã‚‹ index ã‚’ç”Ÿæˆ
292 const std::uint_least64_t index_zyx = static_cast<std::uint_least64_t>((j + z_num) & (z_num - 1)) + index_zy;
293
294 if (tile_cache_.hasAttempted(index_zyx)) {
295 continue;
296 }
297
298 const std::string x_value = std::to_string((j + z_num) & (z_num - 1)); // X ã®å€¤
299
300 // 1. FileTileLoaderï¼ˆãƒ­ãƒ¼ã‚«ãƒ«ãƒ•ã‚¡ã‚¤ãƒ«èª­ã¿è¾¼ã¿ï¼‰
301 auto texture = FileTileLoader::load(
302 local_file_path_zny,
303 x_value
304 );
305
306 if (texture) {
307 tile_cache_.insert(index_zyx, std::move(texture));
308 continue;
309 }
310
311 // 2. UrlTileLoaderï¼ˆURL ã‹ã‚‰ãƒ€ã‚¦ãƒ³ãƒ­ãƒ¼ãƒ‰ï¼‰

316 x_value

322 }
323 }
