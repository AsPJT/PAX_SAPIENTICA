169 bool cellOpen(const AStarNode& node_, std::shared_ptr<Environment>& environment) noexcept {
170 for (std::int_fast32_t x{}; x < 8; ++x) {
171 const AStarVec2 neighbour = AStarVec2{ node_.position.x + astar_adjacent_cell[x].x,
172 node_.position.y + astar_adjacent_cell[x].y };
173 if (neighbour == end_vec2) return true;
174
175 const AStarVec2 neighbour_z = neighbour * z;
176 if (!isRange(neighbour)) continue;
177 //if (!isRange(neighbour) || environment->getSlope(neighbour_z) >= 213) continue;
178 // „Ç≥„Çπ„ÉàË®àÁÆó

180 const GridType distance = calculateDistance(neighbour);
181 if (existPoint(neighbour, node_cost + distance)) continue;
182 open.emplace_back(AStarNode(neighbour, node_.position, distance, node_cost));
183 }
184 return false;
