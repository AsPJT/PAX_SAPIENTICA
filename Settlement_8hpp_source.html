156 double getSNP() const noexcept {
157 std::uint_least64_t snp = 0;
158 for (std::size_t i = 0; i < agents.size(); ++i) {
159 snp += static_cast<std::uint_least64_t>(agents[i].cgetGenome().getSNP());
160 }
161 return static_cast<double>(snp) / static_cast<double>(agents.size()) / 255.0;

163
164 std::uint_least8_t getLanguage() const noexcept {
165 if (agents.empty()) {
166 return 0;
167 }
168
169 std::array<std::uint_least8_t, 256> language{};
170 for (const auto& agent : agents) {
171 ++language[agent.cgetLanguage()];
172 }
173
174 std::size_t max_count = 0;
175 std::size_t max_index = 0;
176 for (std::size_t i = 0; i < language.size(); ++i) {
177 if (language[i] > max_count) {
178 max_count = language[i];
179 max_index = i;
180 }
181 }
182 return static_cast<std::uint_least8_t>(max_index);
183 }
184
185 std::size_t getMostMtDNA() const noexcept {
186 std::size_t mtdna_max = 0;
187 for (std::size_t i = 0; i < agents.size(); ++i) {
188 mtdna_max += agents[i].cgetGenome().getMtDNA();
189 }
190 return mtdna_max / agents.size();
191 }
