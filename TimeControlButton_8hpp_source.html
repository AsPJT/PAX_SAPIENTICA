271 static ButtonType getButtonType(TimeControlButton::Id id) {
272 using Id = TimeControlButton::Id;
273 if (id >= Id::PlaybackReverse && id <= Id::PlaybackForward) return ButtonType::Playback;
274 if (id >= Id::DayBackward && id <= Id::Century100Backward) return ButtonType::Backward;

281 void executeButtonAction(TimeControlButton::Id id) {
282 using Id = TimeControlButton::Id;
283 using Action = TimePlaybackControlEvent::Action;
284
285 // ÂÜçÁîü„Ç≥„É≥„Éà„É≠„Éº„É´
286 if (id == Id::PlaybackReverse) {
287 paxs::EventBus::getInstance().publish(TimePlaybackControlEvent(Action::Reverse));
288 } else if (id == Id::PlaybackStop) {
289 paxs::EventBus::getInstance().publish(TimePlaybackControlEvent(Action::Stop));
290 } else if (id == Id::PlaybackForward) {
291 paxs::EventBus::getInstance().publish(TimePlaybackControlEvent(Action::Forward));
292 } else {
293 // Êó•‰ªòÁßªÂãïÁ≥ª
294 constexpr double day = paxs::Calendar<double>::daysInTropicalDay();
295 constexpr double month = paxs::Calendar<double>::daysInTropicalMonth();
296 constexpr double year = paxs::Calendar<double>::daysInTropicalYear();
297
298 // „Éú„Çø„É≥ID„Åã„ÇâÊó•Êï∞„ÇíË®àÁÆó
299 const double days_map[] = {
300 -day, -month, -year, -year * 10, -year * 100, -year * 1000, -year * 10000, // Backward
301 +day, +month, +year, +year * 10, +year * 100, +year * 1000, +year * 10000 // Forward
302 };
303
304 const int index = static_cast<int>(id) - static_cast<int>(Id::DayBackward);
305 if (index >= 0 && index < 14) {
306 paxs::EventBus::getInstance().publish(DateNavigationEvent(days_map[index]));
307 }
308 }
309 }
