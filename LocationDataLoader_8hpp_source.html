185 static T getOptionalValue(
186 const std::string& str,
187 T default_value
188 ) {
189 if (str.empty()) {
190 return default_value;
191 }
192
193 if constexpr (std::is_same_v<T, std::uint_least16_t>) {

198 PAXS_WARNING("Invalid uint16 value: \"" + str + "\", using default: " + std::to_string(default_value));
199 return default_value;

201 return StringUtils::safeStod(str, default_value, true);
