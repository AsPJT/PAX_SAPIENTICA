41 template <typename T>
42 std::vector<T> select(const std::vector<T>& vec, std::size_t num_elements) noexcept {
43 if (num_elements > vec.size()) {
44 PAXS_ERROR("The number of elements to select is greater than the size of the vector.");
45 return {};
46 }
47
48 std::vector<T> result(vec);
49 for (std::size_t i = 0; i < num_elements; ++i) {
50 std::uniform_int_distribution<std::size_t> distribution(i, result.size() - 1);
51 std::size_t j = distribution(*gen);
52 std::swap(result[i], result[j]);
53 }
54
55 result.resize(num_elements);
56 return result;
57 }

72 std::size_t index_1 = getRandomIndex(num_elements_1, used_indices_1);
73 std::size_t index_2 = getRandomIndex(num_elements_2, used_indices_2);

93 std::uniform_int_distribution<std::size_t> distribution(0, num_elements - 1);
