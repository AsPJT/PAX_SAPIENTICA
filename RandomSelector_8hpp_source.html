38 std::vector<T> select(const std::vector<T>& vec, std::size_t num_elements) noexcept {
39 if (num_elements > vec.size()) {

41 return {};

44 std::vector<T> result(vec);

46 std::uniform_int_distribution<std::size_t> distribution(i, result.size() - 1);
47 std::size_t j = distribution(gen);
48 std::swap(result[i], result[j]);
