112 void handleLogEvent(const LogEvent& event) {
113 while (notification_stack_.size() >= MAX_NOTIFICATIONS) {
114 notification_stack_.pop_front();
115 }
116
117 NotificationItem item;
118 item.display_text = formatMessage(event);
119 item.level = event.level;
120 item.creation_time = getCurrentTime();
121 item.auto_dismiss_duration = AUTO_DISMISS_DURATION;
122 item.id = next_notification_id_++;
123 item.is_close_button_hovered = false;
124
125 notification_stack_.emplace_back(item);
126 updateLayout();
127 }

144 void update(double current_time) {
145 auto it = notification_stack_.begin();
146 while (it != notification_stack_.end()) {
147 if (current_time - it->creation_time >= it->auto_dismiss_duration) {
148 it = notification_stack_.erase(it);
149 } else {
150 ++it;
151 }
152 }
153
154 if (!notification_stack_.empty()) {
155 updateLayout();
156 }

197 std::string formatMessage(const LogEvent& event) const {
198 std::string level_str;
199 switch(event.level) {
200 case LogEvent::Level::Info: level_str = "[INFO]"; break;
201 case LogEvent::Level::Warning: level_str = "[WARNING]"; break;
202 case LogEvent::Level::Error: level_str = "[ERROR]"; break;
203 }
204
205 std::string short_filename = event.filename;
206 size_t last_slash = short_filename.find_last_of("/\\");
207 if (last_slash != std::string::npos) {
208 short_filename = short_filename.substr(last_slash + 1);
209 }
210
211 return level_str + " " + event.message +
212 "\n(" + short_filename + ":" + std::to_string(event.line) + ")";
213 }
