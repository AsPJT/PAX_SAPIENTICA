185 void calculateFPS() {
186 if (frame_times_.empty()) {
187 fps_ = 0.0;
188 frame_time_ms_ = 0.0;
189 return;
190 }
191
192 // Âπ≥Âùá„Éï„É¨„Éº„É†ÊôÇÈñì„ÇíË®àÁÆó
193 double total_time = 0.0;
194 for (double ft : frame_times_) {
195 total_time += ft;
196 }
197 double avg_frame_time = total_time / frame_times_.size();
198
199 frame_time_ms_ = avg_frame_time * 1000.0;
200 fps_ = (avg_frame_time > 0.0) ? (1.0 / avg_frame_time) : 0.0;

300 int renderScopeTree(int pos_x, int current_y, int max_y, const std::string& scope_name, int depth, paxg::Font* font_text) const {
301 if (current_y >= max_y) return current_y;
302
303 const auto& profiler = PerformanceProfiler::getInstance();
304 const auto* scope_data = profiler.getScopeData(scope_name);
305 if (scope_data == nullptr) return current_y;
306
307 // „Ç§„É≥„Éá„É≥„Éà
308 const int indent = PADDING + depth * 12;
309
310 // „Çπ„Ç≥„Éº„ÉóÂêç„Å®ÊôÇÈñì„ÇíÊèèÁîª
311 std::string prefix = (depth > 0) ? " " : "‚ñº ";
312 for (int i = 1; i < depth; ++i) {
313 prefix = " " + prefix;
314 }
315 if (depth > 0) prefix = "‚Ä¢" + prefix;
316
317 const double percentage = (profiler.getTotalFrameTime() > 0.0)
318 ? (scope_data->current_time_ms / profiler.getTotalFrameTime() * 100.0)
319 : 0.0;
320
321 std::string text = prefix + scope_name + " "
322 + formatDouble(scope_data->current_time_ms, 2) + "ms ("
323 + formatDouble(percentage, 0) + "%)";
324
325 paxg::Color text_color = getTimeColor(scope_data->current_time_ms);
326 font_text->draw(text,
327 paxg::Vec2i(pos_x + indent, current_y),
328 text_color);
329
330 current_y += LINE_HEIGHT;
331
332 // Â≠ê„Çπ„Ç≥„Éº„Éó„ÇíÊèèÁîª
333 for (const auto& child_name : scope_data->children) {
334 if (current_y >= max_y) break;
335 current_y = renderScopeTree(pos_x, current_y, max_y, child_name, depth + 1, font_text);
336 }
337
338 return current_y;
339 }

342 void renderFlatMode(Vector2<int> pos, int content_y, paxg::Font* font_text) const {
343 const auto& profiler = PerformanceProfiler::getInstance();
344 const auto& all_scopes = profiler.getAllScopes();
345
346 // „Çπ„Ç≥„Éº„Éó„ÇíÊôÇÈñìÈ†Ü„Å´„ÇΩ„Éº„Éà
347 std::vector<const PerformanceScopeData*> sorted_scopes;
348 for (const auto& pair : all_scopes) {
349 sorted_scopes.push_back(&pair.second);
350 }
351
352 std::sort(sorted_scopes.begin(), sorted_scopes.end(),
353 [](const PerformanceScopeData* a, const PerformanceScopeData* b) {
354 return a->current_time_ms > b->current_time_ms;
355 });
356
357 // ‰∏ä‰Ωç„Çπ„Ç≥„Éº„Éó„ÇíÊèèÁîª
358 int current_y = content_y;
359 const int max_y = pos.y + MONITOR_HEIGHT - PADDING;
360 const std::size_t max_items = (max_y - current_y) / LINE_HEIGHT;
361
362 for (std::size_t i = 0; i < sorted_scopes.size() && i < max_items; ++i) {
363 const auto* scope_data = sorted_scopes[i];
364
365 const double percentage = (profiler.getTotalFrameTime() > 0.0)
366 ? (scope_data->current_time_ms / profiler.getTotalFrameTime() * 100.0)
367 : 0.0;
368
369 std::string text = scope_data->name + " "
370 + formatDouble(scope_data->current_time_ms, 2) + "ms ("
371 + formatDouble(percentage, 0) + "%)";
372
373 paxg::Color text_color = getTimeColor(scope_data->current_time_ms);
374 font_text->draw(text,
375 paxg::Vec2i(pos.x + PADDING, current_y),
376 text_color);
377
378 current_y += LINE_HEIGHT;
379 }
380 }
