121 static int calculateTextLines(const std::string& text, int max_width) {
122 if (text.empty()) {
123 return 0;
124 }
125
126 const int avg_char_width = 12;
127 const int max_chars_per_line = max_width / avg_char_width;
128
129 int line_count = 0;
130 std::string remaining_text = text;
131
132 while (!remaining_text.empty()) {
133 std::size_t split_pos = max_chars_per_line;
134
135 if (remaining_text.length() <= split_pos) {
136 line_count++;
137 break;
138 }
139
140 std::size_t last_space = remaining_text.rfind(' ', split_pos);
141 std::size_t last_comma = remaining_text.rfind(',', split_pos);
142 std::size_t last_jp_comma = remaining_text.rfind(reinterpret_cast<const char*>(u8"„ÄÅ"), split_pos);
143
144 std::size_t actual_split = split_pos;
145 if (last_space != std::string::npos && last_space > split_pos / 2) {
146 actual_split = last_space + 1;
147 } else if (last_comma != std::string::npos && last_comma > split_pos / 2) {
148 actual_split = last_comma + 1;
149 } else if (last_jp_comma != std::string::npos && last_jp_comma > split_pos / 2) {
150 actual_split = last_jp_comma + 3;
151 }
152
153 line_count++;
154 remaining_text = remaining_text.substr(actual_split);
155 }
156
157 return line_count;
158 }
