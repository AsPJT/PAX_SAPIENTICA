100 static bool loadAllKeyValueTSV(const std::string& file_path,
101 paxs::UnorderedMap<std::uint_least32_t, std::string>& map) {
102 std::ifstream file(file_path);
103 if (!file.is_open()) {
104 return false;
105 }
106
107 std::string line;
108 // 1Ë°åÁõÆÔºà„Éò„ÉÉ„ÉÄ„ÉºÔºâ„ÇíË™≠„ÅøËæº„Åø
109 if (!std::getline(file, line)) {
110 return false;
111 }
112
113 // BOMÂâäÈô§
114 if (line.size() >= BOMConstants::bom_size &&
115 static_cast<unsigned char>(line[0]) == BOMConstants::bom_byte_1 &&
116 static_cast<unsigned char>(line[1]) == BOMConstants::bom_byte_2 &&
117 static_cast<unsigned char>(line[2]) == BOMConstants::bom_byte_3) {
118 line = line.substr(BOMConstants::bom_size);
119 }
120
121 std::vector<std::string> headers = paxs::StringUtils::split(line, '\t');
122
123 // key, value „Ç´„É©„É†„ÅÆ„Ç§„É≥„Éá„ÉÉ„ÇØ„Çπ„ÇíÂèñÂæó
124 std::size_t key_index = SIZE_MAX;
125 std::size_t value_index = SIZE_MAX;
126 for (std::size_t i = 0; i < headers.size(); ++i) {
127 if (headers[i] == "key") {
128 key_index = i;
129 }
130 if (headers[i] == "value") {
131 value_index = i;
132 }
133 }
134
135 if (key_index == SIZE_MAX || value_index == SIZE_MAX) {
136 PAXS_ERROR("TSV file is missing 'key' or 'value' column: " + file_path);
137 return false;
138 }
139
140 // „Éá„Éº„ÇøË°å„ÇíË™≠„ÅøËæº„Åø
141 while (std::getline(file, line)) {
142 std::vector<std::string> columns = paxs::StringUtils::split(line, '\t');
143
144 if (key_index >= columns.size() || value_index >= columns.size()) {
145 continue;
146 }
147
148 const std::string& key_str = columns[key_index];
149 const std::string& value_str = columns[value_index];
150
151 if (key_str.empty()) {
152 continue;
153 }
154
155 map.emplace(
156 MurMur3::calcHash(key_str.size(), key_str.c_str()),
157 value_str
158 );
159 }
160
161 return true;
162 }

165 static std::string loadKeyValueTSV(const std::string& file_path, std::uint_least32_t target_key) {
166 paxs::UnorderedMap<std::uint_least32_t, std::string> temp_map;
167 if (!loadAllKeyValueTSV(file_path, temp_map)) {
168 return std::string{};
169 }
170 return temp_map.contains(target_key) ? temp_map.at(target_key) : std::string{};
171 }
