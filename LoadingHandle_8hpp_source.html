163 template <class ResultT, class TaskFunc>
164 LoadingHandle<ResultT> startAsyncTask(TaskFunc&& task) {
165 using State = LoadingState<ResultT>;
166
167 auto state = std::make_shared<State>();
168
169 std::thread worker([state, task = std::forward<TaskFunc>(task)]() mutable {
170 ProgressToken<ResultT> token(state);
171 try {
172 // å®Ÿè¡Œé–‹å§‹ / Start execution
173 token.setProgress(0.0f);
174 token.setMessage("Loading...");
175
176 ResultT result = task(token);
177
178 if (!state->cancelled.load(std::memory_order_relaxed)) {
179 token.setResult(std::move(result));
180 token.setProgress(1.0f);
181 }
182 } catch (...) {
183 state->has_error.store(true, std::memory_order_relaxed);
184 state->error = std::current_exception();
185 }
186
187 state->finished.store(true, std::memory_order_release);
188 });
189
190 return LoadingHandle<ResultT>(std::move(state), std::move(worker));
191 }
