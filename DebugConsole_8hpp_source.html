204 void handleLogEvent(const LogEvent& event) {
205 while (log_history_.size() >= MAX_LOG_HISTORY) {
206 log_history_.pop_front();
207 }
208
209 ConsoleLogItem item;
210 item.display_text = formatMessage(event);

214 log_history_.emplace_back(item);

349 std::string formatMessage(const LogEvent& event) const {
350 std::string level_str;
351 switch(event.level) {
352 case LogEvent::Level::Info: level_str = "[INFO] "; break;
353 case LogEvent::Level::Warning: level_str = "[WARNING]"; break;
354 case LogEvent::Level::Error: level_str = "[ERROR] "; break;
355 }
356
357 std::string short_filename = event.filename;
358 size_t last_slash = short_filename.find_last_of("/\\");
359 if (last_slash != std::string::npos) {
360 short_filename = short_filename.substr(last_slash + 1);
361 }
362
363 return level_str + " " + event.message +
364 " (" + short_filename + ":" + std::to_string(event.line) + ")";
365 }
